Edge Function: hyper-handler (Updated)

import { serve } from "https://deno.land/std@0.168.0/http/server.ts"
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'

serve(async (req) => {
  const { type, webinar_id, name, email, whatsapp_number } = await req.json()
  
  const supabase = createClient(
    Deno.env.get('SUPABASE_URL') ?? '',
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY') ?? ''
  )

  if (type === 'registration') {
    // Store user registration
    const { data: registration, error } = await supabase
      .from('user_registrations')
      .insert({
        webinar_id,
        name,
        email,
        whatsapp_number,
        registration_type: 'live_join'
      })
      .select()
      .single()

    if (error) throw error

    // Get or create Google Meet session
    const { data: meetSession } = await supabase
      .from('google_meet_sessions')
      .select('meet_url')
      .eq('webinar_id', webinar_id)
      .single()

    const meetUrl = meetSession?.meet_url || await generateMeetUrl(webinar_id)
    
    // Sync to Google Sheets
    await syncToGoogleSheets(registration)
    
    return new Response(JSON.stringify({ 
      success: true, 
      meet_url: meetUrl 
    }))
  }
})
